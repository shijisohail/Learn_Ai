# pip install anthropic fastapi import anthropic from typing import Callable import asyncio client = anthropic.Anthropic() HIGH_RISK_TOOLS = {"delete_record", "send_email", "execute_sql", "deploy_code"} class ApprovalGate: def __init__(self, approver: Callable[[str, dict], bool]): """approver(tool_name, tool_args) -> True/False""" self.approver = approver self.audit_log = [] def check_and_execute(self, tool_name: str, tool_args: dict, tool_fn: Callable): """Gate: approve if high-risk, else execute directly.""" if tool_name in HIGH_RISK_TOOLS: # TODO: call self.approver, log decision, execute or raise ... else: # low-risk: execute immediately return tool_fn(**tool_args) def cli_approver(tool_name: str, tool_args: dict) -> bool: """Prompt human via stdin for approval.""" print(f"\n⚠ HIGH-RISK TOOL: {tool_name}") print(f" Args: {tool_args}") answer = input("Approve? [y/N]: ").strip().lower() return answer == "y"
Your Task

Complete ApprovalGate.check_and_execute():

  • For high-risk tools: call self.approver(tool_name, tool_args)
  • Append to self.audit_log a dict with tool, args, approved, timestamp
  • If approved: execute tool_fn(**tool_args) and return result
  • If denied: raise PermissionError("Tool execution denied by human operator")
  • Write a test with an auto-approver lambda that always returns True and one that always returns False
Test Cases
T1

T2

gate.check_and_execute("delete_record", {"id":1}, ...)
→ raises PermissionError
T3
gate.check_and_execute("get_record", {"id":1}, lambda id: "data")
# "get_record" is NOT in HIGH_RISK_TOOLS
→ "data" (no approval needed)
T4
len(gate.audit_log) > 0 and "timestamp" in gate.audit_log[0]
→ True (all decisions logged)
Progressive Hints
