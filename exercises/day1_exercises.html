import anthropic # Official Anthropic Python SDK client = anthropic.Anthropic() # Reads ANTHROPIC_API_KEY from env def count_tokens(text: str) -> dict: """Count tokens WITHOUT a full API call.""" # count_tokens is a special endpoint — no LLM inference, just tokenization response = client.messages.count_tokens( model="claude-haiku-4-5-20251001", # cheapest model messages=[{"role": "user", "content": text}] ) input_tokens = response.input_tokens # The count we care about cost = (input_tokens / 1_000_000) * 0.25 # $0.25 per 1M return { "text": text, "tokens": input_tokens, "cost_usd": cost, "cost_display": f"${cost:.6f}" }
Key insight: The count_tokens endpoint runs the tokenizer only — no inference. It costs roughly 10x less than a full call and lets you pre-flight requests to enforce budgets before spending real money.

The Stateless API — How Memory Works

What YOU Store (messages list)

messages = []

What Model Sees (per API call)

Why "fresh" forgets: If you create a new messages list (or don't pass history), the model has zero context. LLMs have NO server-side memory — every call starts fresh unless YOU send the history.
The chat() Function — Anatomy
┌─ chat(messages, user_input) ──────────────────────────────────┐
1.
APPEND user message to messages list

3.
EXTRACT reply text
reply = response.content[0].text
4.
APPEND assistant reply to messages list
messages = messages + [{"role": "assistant", "content": reply}]
5.
RETURN reply + updated messages
return reply, messages

import anthropic client = anthropic.Anthropic() def run_agent(query: str, tools: list, tool_map: dict) -> str: messages = [{"role": "user", "content": query}] while True: # loop until end_turn or error response = client.messages.create( model="claude-haiku-4-5-20251001", max_tokens=1024, tools=tools, # list of tool schemas messages=messages ) # EXIT CONDITION: model is done if response.stop_reason == "end_turn": return response.content[0].text # TOOL EXECUTION: find all tool_use blocks tool_results = [] for block in response.content: if block.type == "tool_use": fn = tool_map[block.name] # look up Python function result = fn(**block.input) # call it with Claude's args tool_results.append({ "type": "tool_result", "tool_use_id": block.id, "content": str(result) }) # APPEND both model response and tool results to history messages.append({"role": "assistant", "content": response.content}) messages.append({"role": "user", "content": tool_results})
