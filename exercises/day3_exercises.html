Chunking Strategies

▼
Visual Explainer

Same text, 3 different chunking strategies

Recursive Chunker — Decision Tree
text (2000 chars) │ ├── len(text) < chunk_size? → return [text] ✓ base case │ └── Find best split point (priority order): 1. "\n\n" paragraph break ← preferred ⭐ 2. "\n" line break 3. ". " sentence boundary 4. " " word boundary 5. [idx] hard split at chunk_size ← last resort ⚠️ │ ├── left = text[:split_idx] ├── right = text[split_idx - overlap:] ← 10% overlap added └── return [left] + recurse(right)

Your Task — Implement 3 Chunking Functions

  • chunk_fixed(text, size=200) — split at exact character count (ignores words)
  • chunk_sentences(text) — split on . ? ! boundaries, keep sentences whole
  • chunk_recursive(text, size=512, overlap=0.1) — production-grade recursive splitter
  • evaluate(text) — compare all 3: chunk_count, avg_size, has_split_words
Starter Code
import re def chunk_fixed(text: str, size: int = 200) -> list[str]: """Split text into fixed character-size chunks. May break mid-word.""" # TODO: return [text[i:i+size] for i in range(0, len(text), size)] ... def chunk_sentences(text: str) -> list[str]: """Split on sentence boundaries (. ? !). Group into ~200-char windows.""" # Hint: re.split(r'(?<=[.?!])\s+', text) ... def chunk_recursive(text: str, size: int = 512, overlap: float = 0.1) -> list[str]: """Recursive character splitter — tries separators in priority order.""" separators = ["\n\n", "\n", ". ", " "] if len(text) <= size: return [text] overlap_chars = int(size * overlap) # TODO: try each separator, find split point near middle # TODO: split into left + right (with overlap), recurse on right ... def evaluate(text: str) -> dict: """Compare all 3 strategies on the same text.""" results = {} for name, fn in [("fixed", chunk_fixed), ("sentence", chunk_sentences), ("recursive", chunk_recursive)]: chunks = fn(text) results[name] = { "count": len(chunks), "avg_size": sum(len(c) for c in chunks) // max(1, len(chunks)), "split_words": sum(1 for c in chunks if c[-1:].isalpha() and len(c) == 200) } return results

Your Task — Build RAGPipeline

  • ingest(texts: list[str]) — chunk each text, embed chunks, store in-memory vector store
  • query(question: str, k=3) -> str — embed question, retrieve top-k chunks, build prompt, call LLM
  • add_document(text: str) — add single document (calls ingest internally)
  • stats() -> dict — return {total_chunks, total_embeddings, documents_ingested}
Starter Code
from dataclasses import dataclass import numpy as np import anthropic @dataclass class Chunk: text: str embedding: list[float] doc_id: int class RAGPipeline: def __init__(self): self.client = anthropic.Anthropic() self.chunks: list[Chunk] = [] self._doc_count = 0 self.cache = EmbeddingCache() # from Exercise 1 def ingest(self, texts: list[str]) -> None: for text in texts: chunks = chunk_recursive(text) # from Exercise 2 for c in chunks: embedding = self.cache.embed(c) self.chunks.append(Chunk(c, embedding, self._doc_count)) self._doc_count += 1 def _retrieve(self, query: str, k: int) -> list[Chunk]: # TODO: embed query, compute cosine sim with all chunks, return top-k ... def query(self, question: str, k: int = 3) -> str: chunks = self._retrieve(question, k) context = "\n".join(f"[{i+1}] {c.text}" for i, c in enumerate(chunks)) prompt = f"""CONTEXT:\n{context}\n\nQUESTION: {question}""" # TODO: call self.client.messages.create() ... def add_document(self, text: str) -> None: self.ingest([text]) def stats(self) -> dict: return { "total_chunks": len(self.chunks), "total_embeddings": len(self.chunks), "documents_ingested": self._doc_count }

Your Task — Build MemoryAgent (all 3 tiers)

  • chat(user_msg) — process message, maintain memory, return response
  • L1: keep last 5 turns in messages[] (always in context)
  • L2: when L1 overflows, summarize old turns with Claude → store as episode_summary
  • L3: embed every turn, store in vector_memory[] — retrieval by similarity
  • memory_status() — return current token counts for all 3 tiers
Starter Code
from dataclasses import dataclass, field from typing import Optional import anthropic @dataclass class Turn: role: str # "user" | "assistant" content: str turn_num: int embedding: Optional[list[float]] = None class MemoryAgent: L1_MAX_TURNS = 5 def __init__(self): self.client = anthropic.Anthropic() self.cache = EmbeddingCache() # L1: Active context (recent turns) self.active: list[Turn] = [] # L2: Episode summary (compressed older turns) self.episode_summary: str = "" # L3: Vector memory (all turns ever) self.vector_memory: list[Turn] = [] self._turn_count = 0 def _compress_to_l2(self) -> None: """When L1 is full: summarize oldest turns → episode_summary.""" old_turns = self.active[:self.L1_MAX_TURNS // 2] self.active = self.active[self.L1_MAX_TURNS // 2:] turns_text = "\n".join(f"{t.role}: {t.content}" for t in old_turns) # TODO: call Claude to summarize old_turns, append to self.episode_summary ... def _retrieve_from_l3(self, query: str, k: int = 3) -> list[Turn]: """Find most relevant past turns by cosine similarity.""" # TODO: embed query, score all vector_memory turns, return top-k ... def _build_context(self, user_msg: str) -> list[dict]: """Assemble full context: L3 retrieved + L2 summary + L1 active.""" messages = [] # Retrieve relevant past context (L3) relevant = self._retrieve_from_l3(user_msg) if relevant: mem_text = "Relevant past context:\n" + "\n".join( f"[Turn {t.turn_num}] {t.role}: {t.content}" for t in relevant ) messages.append({"role": "user", "content": mem_text}) messages.append({"role": "assistant", "content": "Understood."}) # Add L2 summary if exists if self.episode_summary: messages.append({"role": "user", "content": f"Summary: {self.episode_summary}"}) messages.append({"role": "assistant", "content": "Understood."}) # Add L1 active turns for t in self.active: messages.append({"role": t.role, "content": t.content}) return messages def chat(self, user_msg: str) -> str: self._turn_count += 1 # 1. Check if L1 compression needed if len(self.active) >= self.L1_MAX_TURNS * 2: self._compress_to_l2() # 2. Build context and call Claude messages = self._build_context(user_msg) messages.append({"role": "user", "content": user_msg}) response = self.client.messages.create( model="claude-opus-4-5", max_tokens=512, system="You are a helpful assistant with long-term memory.", messages=messages ) assistant_msg = response.content[0].text # 3. Store in L1 and L3 user_turn = Turn("user", user_msg, self._turn_count) asst_turn = Turn("assistant", assistant_msg, self._turn_count) self.active.extend([user_turn, asst_turn]) # Embed and store in L3 user_turn.embedding = self.cache.embed(user_msg) self.vector_memory.append(user_turn) return assistant_msg
