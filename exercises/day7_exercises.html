🧬 Code Anatomy — Regex Patterns
Pattern Matches Example
^diff --git New file boundary — reset current file context diff --git a/auth.py b/auth.py
^\+\+\+ b/(.+) Capture the modified filename (group 1) +++ b/auth.py → "auth.py"
^@@ -\d+,\d+ \+(\d+) Hunk header — extract start line number from group 1 @@ -10,6 +10,8 @@ → line 10
^\+(?!\+\+) Added line (exclude +++ file header) + return db.execute(query)
^-(?!--) Removed line (exclude --- file header) - query = "..." + user_id
^ Context line (space prefix, unchanged) def authenticate(user_id):
Python
import re from dataclasses import dataclass, field from typing import Optional @dataclass class ParsedDiff: file: str added_lines: list[str] = field(default_factory=list) removed_lines: list[str] = field(default_factory=list) context_lines: list[str] = field(default_factory=list) start_line: int = 0 def parse_pr_diff(diff_text: str) -> list[ParsedDiff]: results: list[ParsedDiff] = [] current: Optional[ParsedDiff] = None for line in diff_text.splitlines(): # New file boundary if line.startswith("diff --git"): if current: results.append(current) current = None # Capture filename from +++ b/path elif m := re.match(r'^\+\+\+ b/(.+)', line): current = ParsedDiff(file=m.group(1)) # Hunk header — extract start line elif current and (m := re.match(r'^@@ -\d+,\d+ \+(\d+)', line)): current.start_line = int(m.group(1)) # Added line (exclude +++ file header) elif current and re.match(r'^\+(?!\+\+)', line): current.added_lines.append(line[1:]) # strip the leading + # Removed line (exclude --- file header) elif current and re.match(r'^-(?!--)', line): current.removed_lines.append(line[1:]) # Context line elif current and line.startswith(" "): current.context_lines.append(line[1:]) if current: results.append(current) return results

💡 Hints (click to reveal) ▾ Show
1. Use the walrus operator := (Python 3.8+) to match and capture in the same line — much cleaner than two-line if/match.
2. The tricky part: +++ b/filename and --- a/filename lines start with + and - but aren't code changes. Use ^\+\+\+ and ^--- as negative lookahead.
3. Don't forget to append the final current after the loop — the last file has no subsequent diff --git to trigger append.
4. Strip the leading marker character before storing: line[1:] removes the + or - prefix.

🧬 Code Anatomy — Forced Structured Output
Python — Forced Tool Use Pattern
from pydantic import BaseModel from typing import Literal import anthropic class SecurityFinding(BaseModel): severity: Literal["CRITICAL", "HIGH", "MEDIUM", "LOW"] title: str line: int code: str # the offending code snippet fix: str # concrete fix suggestion class SecurityReport(BaseModel): findings: list[SecurityFinding] # ─── The Trick: define output schema as a "tool" ─────────────── # The model MUST "call" this tool — guaranteeing our JSON schema. # This is more reliable than JSON mode or prompt-based parsing. REPORT_TOOL = { "name": "report_findings", "description": "Report security findings found in the code", "input_schema": SecurityReport.model_json_schema() } client = anthropic.Anthropic() def security_agent(diff: ParsedDiff) -> SecurityReport: code_context = f""" File: {diff.file} (starting at line {diff.start_line}) Added lines (review carefully): {chr(10).join(f' + {l}' for l in diff.added_lines)} Removed lines (for context): {chr(10).join(f' - {l}' for l in diff.removed_lines)} Context: {chr(10).join(f' {l}' for l in diff.context_lines)} """ response = client.messages.create( model="claude-opus-4-5", max_tokens=1024, system="""You are a security expert reviewing code changes. Look for: SQL injection, hardcoded secrets, path traversal, command injection, insecure deserialization. Focus ONLY on security — ignore style and performance.""", tools=[REPORT_TOOL], tool_choice={"type": "tool", "name": "report_findings"}, # ↑ FORCE this specific tool — no other response possible messages=[{"role": "user", "content": code_context}] ) # content[0] is ALWAYS a tool_use block — safe to index directly tool_input = response.content[0].input return SecurityReport.model_validate(tool_input)
Why tool_choice over JSON mode?
JSON mode can produce any JSON. tool_choice: {"type": "tool", "name": "X"} forces the model to output exactly the schema defined in the tool's input_schema. Combined with Pydantic's model_validate(), you get strict runtime type checking. Zero regex parsing. Zero hallucinated fields.

🧬 Deduplication Logic
Python — Critic Agent
from pydantic import BaseModel SEVERITY_ORDER = {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "LOW": 3} def deduplicate_findings( security: list[SecurityFinding], performance: list[PerformanceFinding], style: list[StyleFinding] ) -> list[UnifiedFinding]: all_findings = ( [("security", f) for f in security] + [("performance", f) for f in performance] + [("style", f) for f in style] ) seen: dict[tuple, UnifiedFinding] = {} for category, finding in all_findings: # Duplicate key: same file + line + category = same issue key = (finding.line, finding.title.lower()[:20]) if key not in seen: seen[key] = UnifiedFinding( category=category, **finding.model_dump() ) else: # Keep the higher-severity version existing = seen[key] if SEVERITY_ORDER[finding.severity] < SEVERITY_ORDER[existing.severity]: seen[key] = UnifiedFinding(category=category, **finding.model_dump()) # Sort: CRITICAL first, then HIGH, MEDIUM, LOW return sorted( seen.values(), key=lambda f: SEVERITY_ORDER[f.severity] ) def critic_agent( security: SecurityReport, performance: PerformanceReport, style: StyleReport ) -> list[UnifiedFinding]: # Deterministic deduplication — no LLM call needed here return deduplicate_findings( security.findings, performance.findings, style.findings ) def formatter_agent(findings: list[UnifiedFinding], diff_context: str) -> ReviewReport: # Use LLM for synthesis — executive summary, praise, quick wins summary_prompt = f""" Given these {len(findings)} code review findings: {[f.model_dump() for f in findings]} Write: 1. An executive summary (2-3 sentences) 2. Quick wins (issues fixable in under 5 min) 3. Positive observations about the code 4. Overall risk level: CRITICAL / HIGH / MEDIUM / LOW """ response = client.messages.create( model="claude-opus-4-5", max_tokens=1024, tools=[REVIEW_REPORT_TOOL], tool_choice={"type": "tool", "name": "create_review_report"}, messages=[{"role": "user", "content": summary_prompt}] ) return ReviewReport.model_validate(response.content[0].input)

💡 Hints (click to reveal) ▾ Show
1. Make deduplication deterministic — no LLM. The Critic Agent's deduplication step should be pure Python logic. The LLM is expensive; save it for synthesis (Formatter).
2. Title comparison: use title.lower()[:20] as a fuzzy key — "SQL Injection" and "SQL injection mention" both start with "sql injection". This is intentionally approximate.
3. The Formatter agent should be told the number of CRITICAL vs HIGH issues in its prompt — this helps it calibrate the risk_level without needing to recount.
4. model.model_dump() gives you a plain dict — use it to construct the combined prompt string for the Formatter rather than iterating fields manually.

AgentState TypedDict
from typing import TypedDict, Annotated import operator class AgentState(TypedDict): # Input pr_url: str diffs: list[ParsedDiff] # Parallel outputs — Annotated[list, operator.add] means # each node APPENDS to the list rather than replacing it. # This is the LangGraph fan-in pattern. security_findings: Annotated[list[SecurityFinding], operator.add] perf_findings: Annotated[list[PerformanceFinding], operator.add] style_findings: Annotated[list[StyleFinding], operator.add] # Final outputs unified_findings: list[UnifiedFinding] report: Optional[ReviewReport]
Fan-out with Send() — parallel specialist dispatch
from langgraph.graph import StateGraph, END, Send def dispatch_specialists(state: AgentState): # Send() creates parallel branches — one per specialist per diff # All three run concurrently, results merged via operator.add return [ Send("security_node", {"diff": state["diffs"][0]}), Send("perf_node", {"diff": state["diffs"][0]}), Send("style_node", {"diff": state["diffs"][0]}), ] def security_node(state: dict) -> dict: report = security_agent(state["diff"]) return {"security_findings": report.findings} # ↑ operator.add reducer appends this to existing list # Same pattern for perf_node and style_node
Fan-in — merge results, run Critic + Formatter
def critic_node(state: AgentState) -> dict: # By the time we reach critic_node, all 3 parallel branches # have completed and their findings are merged in state. unified = critic_agent( security=state["security_findings"], performance=state["perf_findings"], style=state["style_findings"] ) return {"unified_findings": unified} def formatter_node(state: AgentState) -> dict: report = formatter_agent( findings=state["unified_findings"], diff_context=f"Reviewing {len(state['diffs'])} files" ) return {"report": report} # Wire the graph graph = StateGraph(AgentState) graph.add_node("fetch", fetch_node) graph.add_node("security_node", security_node) graph.add_node("perf_node", perf_node) graph.add_node("style_node", style_node) graph.add_node("critic", critic_node) graph.add_node("formatter", formatter_node) graph.set_entry_point("fetch") graph.add_conditional_edges("fetch", dispatch_specialists) graph.add_edge("security_node", "critic") graph.add_edge("perf_node", "critic") graph.add_edge("style_node", "critic") graph.add_edge("critic", "formatter") graph.add_edge("formatter", END) app = graph.compile()

💡 Hints (click to reveal) ▾ Show
1. LangGraph's Send() creates parallel branches but the fan-in requires all branches to resolve before the next node runs. The operator.add reducer handles the merge automatically.
2. For the planted bugs test: create the diff manually as a string (don't actually modify files). Paste the complete unified diff format into your POST request body as diff_text.
3. FastAPI's response_model=ReviewReport automatically validates and serializes the response. If the ReviewReport doesn't match, you'll get a 422 — use this to catch Pydantic errors early.
4. Bug 3 (N+1) is a Performance finding, not Security. Your Critic agent must not deduplicate it against the SQL injection. They're on different lines and different categories.
5. For production: add asyncio.gather() timeouts to prevent slow LLM calls from blocking indefinitely. Wrap each specialist in asyncio.wait_for(..., timeout=30).
