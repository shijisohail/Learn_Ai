MODEL_CASCADE = [
    "gpt-4o-mini",
    "claude-haiku-4-5-20251001",
    "gpt-4o",
]
BUDGET = 0.01  # USD per session
session_cost = 0.0

class BudgetExceededError(Exception): pass

def litellm_with_cascade(messages: list[dict]) -> str:
    global session_cost
    for model in MODEL_CASCADE:
        try:
            resp = litellm.completion(model=model, messages=messages)
            cost = litellm.completion_cost(completion_response=resp)
            if session_cost + cost > BUDGET:
                raise BudgetExceededError(f"Budget ${BUDGET:.4f} exceeded. Session total: ${session_cost:.6f}")
            session_cost += cost
            print(f"  [cost] Model={model}, query=${cost:.6f}, session=${session_cost:.6f}")
            return resp.choices[0].message.content
        except BudgetExceededError:
            raise
        except (litellm.exceptions.RateLimitError,
                litellm.exceptions.ServiceUnavailableError) as e:
            print(f"  [fallback] {model} failed: {e}. Trying next...")
            continue
    raise RuntimeError("All models in cascade failed")

    ("system", "Classify the user question as either 'factual' or 'research'.\n"
              "- 'factual': a direct question answerable from a single source (what, how, define)\n"
              "- 'research': requires synthesizing multiple sources, comparing, or deep analysis\n"
              "Respond with ONLY the word 'factual' or 'research', lowercase, no punctuation."),
    ("human", "{question}"),
])

classify_llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
classifier_chain = classify_prompt | classify_llm | StrOutputParser()

def hyde_query(query: str) -> str:
    """Generate a hypothetical document for HyDE retrieval."""
    hyde_prompt = ChatPromptTemplate.from_template(
        "Write a 2-sentence answer to: {query}\nBe factual and technical."
    )
    chain = hyde_prompt | classify_llm | StrOutputParser()
    return chain.invoke({"query": query})

class KnowledgeAssistant:
    def __init__(self):
        self.kb = KnowledgeBase()
        self.reranker = CrossEncoderReranker(top_k=5)
        self.total_cost = 0.0
        self.query_count = 0
        self.latencies = []
        self.cache_hits_start = 0

    def ask(self, question: str) -> dict:
        start = time.time()
        self.query_count += 1
        cost_before = session_cost  # from litellm tracker in ex3

        # Step 1: classify
        query_type = classifier_chain.invoke({"question": question}).strip().lower()
        if query_type not in ("factual", "research"):
            query_type = "factual"  # default to factual on bad output

        if query_type == "factual":
            # Step 2a: HyDE expansion
            hypothesis = hyde_query(question)
            # TODO: use hypothesis (not raw question) for hybrid retrieval
            candidates = hybrid_chain.invoke(hypothesis)[:15]

            # Step 2b: Cross-encoder re-rank
            reranked = self.reranker.invoke({"query": question, "candidates": candidates})

            # Step 2c: Format context and get answer
            context = format_context(reranked)
            messages = [
                {"role": "system", "content": "Answer concisely from the context provided."},
                {"role": "user", "content": f"Context:\n{context}\n\nQuestion: {question}"},
            ]
            answer = litellm_with_cascade(messages)
            sources = [r.get("metadata", {}).get("source", "unknown") for r in reranked[:3]]

        else:  # research
            # TODO: run crew.kickoff(inputs={"topic": question})
            # answer = result.raw; sources = ["crewai-agent"]
            pass

        elapsed = (time.time() - start) * 1000
        self.latencies.append(elapsed)
        query_cost = session_cost - cost_before
        self.total_cost += query_cost

        return {
            "answer": answer,
            "query_type": query_type,
            "sources": sources,
            "cost_usd": round(query_cost, 8),
            "latency_ms": round(elapsed, 1),
        }

    def summary(self):
        avg_lat = sum(self.latencies) / len(self.latencies) if self.latencies else 0
        cache_hits = self.kb.embedder.hits
        cache_total = self.kb.embedder.hits + self.kb.embedder.misses
        hit_rate = cache_hits / cache_total if cache_total else 0

        print(f"""
=== Session Summary ===
Queries:        {self.query_count}
Total cost:     ${self.total_cost:.6f}
Avg latency:    {avg_lat:.0f}ms
Cache hit rate: {hit_rate:.1%} ({cache_hits}/{cache_total})
""")


# Interactive REPL
if __name__ == "__main__":
    assistant = KnowledgeAssistant()
    print("Knowledge Assistant ready. Type 'quit' to exit.\n")

    while True:
        question = input("You: ").strip()
        if question.lower() in ("quit", "exit", "q"):
            break
        if not question:
            continue
        response = assistant.ask(question)
        print(f"\n[{response['query_type'].upper()} | {response['latency_ms']}ms | ${response['cost_usd']:.6f}]")
        print(f"Sources: {', '.join(response['sources'])}")
        print(f"Answer: {response['answer']}\n")

    assistant.summary()
