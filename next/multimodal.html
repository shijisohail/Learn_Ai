PYTHON — Vision API Message Format
import anthropic, base64 client = anthropic.Anthropic() # Method 1: URL-based image (simpler, image must be publicly accessible) response = client.messages.create( model="claude-sonnet-4-5", max_tokens=1024, messages=[{ "role": "user", "content": [ { "type": "image", "source": { "type": "url", "url": "https://example.com/chart.png", }, }, {"type": "text", "text": "What does this chart show? Summarize the trend."} ], }], ) # Method 2: Base64-encoded image (for local files, private images) with open("screenshot.png", "rb") as f: image_data = base64.standard_b64encode(f.read()).decode("utf-8") response = client.messages.create( model="claude-sonnet-4-5", max_tokens=1024, messages=[{ "role": "user", "content": [ { "type": "image", "source": { "type": "base64", "media_type": "image/png", "data": image_data, }, }, {"type": "text", "text": "Describe any bugs or UI issues you see."} ], }], ) print(response.content[0].text)
02 — Document Processing

Document Processing Agent

Extract structured data from PDFs — invoices, forms, reports. The pipeline: PDF → page images → Claude Vision → structured JSON. No OCR library needed.

📄
Input: PDF file
Any PDF — invoice, contract, report. Can be scanned (image-based) or digital.
🖼️
Convert pages to images
pdf2image library: each page becomes a high-res PNG. Resolution matters — 150 DPI for text, 300 DPI for dense tables.
👁️
Send each page to Claude Vision
Pass the page image + extraction schema as instructions. Claude reads layout, tables, checkboxes — things OCR misses.
🔗
Merge & validate across pages
Combine per-page results. Validate required fields are present. Handle multi-page tables that span pages.
✅
Output: Structured JSON
Type-validated Pydantic model. Post to your API or write to database directly.
PYTHON — Invoice Extraction Agent
import asyncio, base64 from pathlib import Path from typing import Optional import anthropic from pdf2image import convert_from_path from pydantic import BaseModel import json, io class InvoiceData(BaseModel): invoice_number: str vendor_name: str date: str total_amount: float line_items: list[dict] tax_amount: Optional[float] = None def pdf_page_to_base64(page_image) -> str: buf = io.BytesIO() page_image.save(buf, format="PNG") return base64.standard_b64encode(buf.getvalue()).decode() EXTRACTION_PROMPT = """Extract all invoice data from this image. Return ONLY valid JSON matching this schema: { "invoice_number": "string", "vendor_name": "string", "date": "YYYY-MM-DD", "total_amount": number, "line_items": [{"description": "str", "quantity": num, "unit_price": num, "total": num}], "tax_amount": number or null } If a field is not visible, use null. Do not add any text outside the JSON.""" async def extract_invoice_page( client: anthropic.AsyncAnthropic, page_b64: str, page_num: int ) -> dict: response = await client.messages.create( model="claude-sonnet-4-5", max_tokens=2048, messages=[{ "role": "user", "content": [ {"type": "image", "source": { "type": "base64", "media_type": "image/png", "data": page_b64 }}, {"type": "text", "text": EXTRACTION_PROMPT}, ], }], ) return json.loads(response.content[0].text) async def extract_invoice(pdf_path: str) -> InvoiceData: client = anthropic.AsyncAnthropic() pages = convert_from_path(pdf_path, dpi=150) # Process all pages concurrently tasks = [ extract_invoice_page(client, pdf_page_to_base64(page), i) for i, page in enumerate(pages) ] page_results = await asyncio.gather(*tasks) # Merge: first page has header, later pages may have more line items merged = page_results[0] for result in page_results[1:]: if result.get("line_items"): merged["line_items"].extend(result["line_items"]) return InvoiceData(**merged) # Usage invoice = asyncio.run(extract_invoice("invoice.pdf")) print(invoice.model_dump_json(indent=2))
03 — Screenshot Analysis

Screenshot Analysis Agent

An agent that takes browser screenshots, analyzes the UI for issues, and produces a structured accessibility/bug report. Useful for automated QA pipelines.

PYTHON — UI Bug Detection Agent
import asyncio, base64 from playwright.async_api import async_playwright from anthropic import AsyncAnthropic from pydantic import BaseModel from typing import Literal import json class UIIssue(BaseModel): severity: Literal["critical", "high", "medium", "low"] category: Literal["accessibility", "layout", "content", "performance"] description: str element: str # CSS selector or description suggestion: str UI_ANALYSIS_PROMPT = """Analyze this screenshot for UI/UX issues. Check for: 1. Accessibility: missing alt text, poor contrast, missing labels 2. Layout: overflow, misalignment, broken responsive design 3. Content: truncated text, missing images, error states visible 4. Usability: confusing hierarchy, unclear CTAs Return JSON array of issues: [{"severity": "critical|high|medium|low", "category": "accessibility|layout|content|performance", "description": "what's wrong", "element": "which element", "suggestion": "how to fix"}] Return empty array [] if no issues found.""" async def analyze_page(url: str) -> list[UIIssue]: client = AsyncAnthropic() issues = [] async with async_playwright() as p: browser = await p.chromium.launch() page = await browser.new_page(viewport={"width": 1280, "height": 800}) await page.goto(url, wait_until="networkidle") # Also check mobile viewport for viewport in [ {"width": 1280, "height": 800}, {"width": 390, "height": 844}, # iPhone 14 Pro ]: await page.set_viewport_size(viewport) screenshot = await page.screenshot(full_page=True) img_b64 = base64.standard_b64encode(screenshot).decode() response = await client.messages.create( model="claude-sonnet-4-5", max_tokens=2048, messages=[{"role": "user", "content": [ {"type": "image", "source": { "type": "base64", "media_type": "image/png", "data": img_b64, }}, {"type": "text", "text": f"Viewport: {viewport['width']}x{viewport['height']}. {UI_ANALYSIS_PROMPT}"}, ]}], ) page_issues = json.loads(response.content[0].text) issues.extend([UIIssue(**i) for i in page_issues]) await browser.close() # Sort by severity severity_order = {"critical": 0, "high": 1, "medium": 2, "low": 3} issues.sort(key=lambda x: severity_order[x.severity]) return issues
04 — Audio Pipeline

Audio with Whisper → Claude Pipeline

Claude doesn't natively process audio yet, but you can build a powerful pipeline: transcribe with OpenAI Whisper, then analyze/summarize with Claude. Extremely useful for meeting notes, podcast analysis, and customer call processing.

PYTHON — Audio Transcription + Analysis Pipeline
import asyncio import openai import anthropic from pathlib import Path from pydantic import BaseModel from typing import Optional class MeetingNotes(BaseModel): summary: str action_items: list[str] decisions: list[str] participants_mentioned: list[str] follow_up_date: Optional[str] async def transcribe_audio(audio_path: str) -> str: """Transcribe using OpenAI Whisper API.""" client = openai.AsyncOpenAI() with open(audio_path, "rb") as audio_file: transcript = await client.audio.transcriptions.create( model="whisper-1", file=audio_file, response_format="verbose_json", # includes timestamps + segments language="en", ) return transcript.text async def analyze_transcript(transcript: str, context: str = "") -> MeetingNotes: """Analyze transcript with Claude and extract structured notes.""" claude = anthropic.AsyncAnthropic() prompt = f"""Analyze this meeting transcript and extract structured information. {f"Context: {context}" if context else ""} TRANSCRIPT: {transcript} Return JSON with this exact structure: {{ "summary": "2-3 sentence executive summary", "action_items": ["who: what by when", ...], "decisions": ["decision made", ...], "participants_mentioned": ["name", ...], "follow_up_date": "YYYY-MM-DD or null" }}""" response = await claude.messages.create( model="claude-sonnet-4-5", max_tokens=2048, messages=[{"role": "user", "content": prompt}], ) import json data = json.loads(response.content[0].text) return MeetingNotes(**data) async def process_meeting_recording(audio_path: str) -> MeetingNotes: print("Transcribing audio...") transcript = await transcribe_audio(audio_path) print(f"Transcript length: {len(transcript.split())} words") print("Analyzing with Claude...") notes = await analyze_transcript(transcript) return notes # Usage notes = asyncio.run(process_meeting_recording("meeting.mp3")) print("Summary:", notes.summary) print("Action items:", notes.action_items)
05 — Production

Production Considerations

Vision APIs are powerful but have meaningful cost and latency implications. Optimize before scaling to production.

Model Cost per Image Notes
Claude Haiku 3.5 ~$0.0006 Fast, cheap. Good for simple classification
Claude Sonnet 4.5 ~$0.0048 Best quality/cost. Recommended for extraction
GPT-4o ~$0.0051 Comparable to Sonnet. Different strengths
Whisper (audio) $0.006/min Per minute of audio, not per request
💰
Image sizing: Claude counts image tokens by resolution. A 1568×1568 image costs ~1600 tokens. Resize images before sending — most use cases don't need full resolution. Target 800px on the longer edge for text/document analysis.
PYTHON — Image Optimization Before API Call
from PIL import Image import io, base64 def optimize_image( image_path: str, max_dimension: int = 1024, quality: int = 85, ) -> str: """Resize + compress image, return base64. Reduces token cost significantly.""" img = Image.open(image_path) # Convert RGBA to RGB if needed if img.mode == "RGBA": img = img.convert("RGB") # Resize if larger than max_dimension if max(img.size) > max_dimension: ratio = max_dimension / max(img.size) new_size = (int(img.width * ratio), int(img.height * ratio)) img = img.resize(new_size, Image.LANCZOS) buf = io.BytesIO() img.save(buf, format="JPEG", quality=quality, optimize=True) return base64.standard_b64encode(buf.getvalue()).decode() # Cache image analysis results (images don't change) import hashlib from functools import lru_cache def image_hash(image_bytes: bytes) -> str: return hashlib.sha256(image_bytes).hexdigest() # In production: use Redis cache with image hash as key # cache[image_hash] = analysis_result # If same invoice uploaded twice, skip the API call
