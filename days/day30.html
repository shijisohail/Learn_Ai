from fastapi import FastAPI, Depends, HTTPException from langgraph.graph import StateGraph, END import instructor from anthropic import Anthropic from qdrant_client import QdrantClient from qdrant_client.models import Filter, FieldCondition, MatchValue import jwt, json, re # ── LAYER 0: Setup ─────────────────────────────────────────────── app = FastAPI(title="AI Copilot Capstone") client = instructor.from_anthropic(Anthropic()) qdrant = QdrantClient("http://qdrant:6333") # ── LAYER 1: JWT Auth + Tenant Extraction (Day 25) ─────────────── def get_tenant(token: str) -> dict: payload = jwt.decode(token, "SECRET", algorithms=["HS256"]) return {"tenant_id": payload["tenant_id"], "roles": payload.get("roles", [])} # ── LAYER 2: Guardrails (Day 23) ───────────────────────────────── INJECTION_RE = re.compile(r"ignore.*(previous|prior).*(instruction|rule)", re.I) def check_guardrails(text: str): if INJECTION_RE.search(text): raise HTTPException(400, "Input rejected by guardrails layer 1") # ── LAYER 3: Intent Classification (Day 29) ────────────────────── def classify_intent(text: str) -> str: resp = Anthropic().messages.create( model="claude-3-5-haiku-20241022", max_tokens=30, system='Respond ONLY: question|action|search|smalltalk', messages=[{"role":"user","content":text}] ) return resp.content[0].text.strip() # ── LAYER 4: Tenant-Scoped RAG (Days 12+25) ────────────────────── def rag_search(query: str, tenant_id: str, vector) -> list: return qdrant.search( collection_name="documents", query_vector=vector, query_filter=Filter(must=[FieldCondition( key="metadata.tenant_id", match=MatchValue(value=tenant_id) )]), limit=5, with_payload=True ) # ── LAYER 5: Structured Response (Day 24) ──────────────────────── from pydantic import BaseModel, Field class CopilotResponse(BaseModel): answer: str sources: list[str] confidence: float = Field(ge=0, le=1) requires_human_review: bool # Day 24 model="claude-3-5-sonnet-20241022", max_tokens=1024, response_model=CopilotResponse, system=f"You are a helpful assistant.
