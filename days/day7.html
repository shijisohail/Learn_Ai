Section 02

Core Implementation

The complete capstone agent — integrating LangGraph state management, specialized Claude subagents, RAG for best-practice retrieval, and structured output for the final review report.

code_review_agent.py
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.sqlite import SqliteSaver
from typing import TypedDict, Annotated
import operator, anthropic, chromadb, voyageai, json

# ─── STATE ───────────────────────────────────────
class ReviewState(TypedDict):
    code: str                   # the code to review
    language: str               # "python", "javascript", etc.
    best_practices: list[str]  # retrieved from knowledge base
    bug_findings: str           # from bug_detector_node
    security_findings: str      # from security_node
    style_findings: str         # from style_node
    final_report: dict          # structured output

claude  = anthropic.Anthropic()
voyage  = voyageai.Client()
chroma  = chromadb.PersistentClient("./codebase_kb")
kb      = chroma.get_or_create_collection("python_best_practices")

# ─── NODE 1: RAG Retrieval ────────────────────────
def retrieve_practices(state: ReviewState) -> dict:
    """Find relevant best practices for the code being reviewed."""
    q = f"Python best practices for {state.get('language','python')} code quality"
    q_vec = voyage.embed([q], model="voyage-3", input_type="query").embeddings
    results = kb.query(query_embeddings=q_vec, n_results=4,
                        include=["documents", "distances"])
    docs = [d for d, s in zip(results["documents"][0], results["distances"][0])
            if (1-s) > 0.6]
    return {"best_practices": docs}

# ─── NODE 2: Bug Detector (Haiku — fast) ─────────
def detect_bugs(state: ReviewState) -> dict:
    practices = "\n".join(state["best_practices"])
    resp = claude.messages.create(
        model="claude-3-5-haiku-20241022", max_tokens=1024,
        system=f"""You are a Python bug detection specialist.
Find ALL bugs: logic errors, off-by-one errors, exception handling issues,
resource leaks, unhandled edge cases.
Reference these best practices when relevant:
{practices}

Format each bug as:
LINE: [line number or range]
TYPE: [error type]
SEVERITY: CRITICAL | HIGH | MEDIUM | LOW
ISSUE: [description]
FIX: [corrected code]""",
        messages=[{"role": "user", "content": f"Review this code:\n```python\n{state['code']}\n```"}]
    )
    return {"bug_findings": resp.content[0].text}

# ─── NODE 3: Security Analyzer (Sonnet — thorough) ─
def analyze_security(state: ReviewState) -> dict:
    resp = claude.messages.create(
        model="claude-sonnet-4-5", max_tokens=1024,
        system="""You are an application security expert (OWASP, CWE knowledge).
Find ALL security vulnerabilities: injection risks (SQL, command, SSTI),
auth/authz issues, hardcoded secrets, insecure deserialization,
path traversal, unsafe eval/exec, missing input validation.
Classify each finding with CVE/CWE reference where applicable.""",
        messages=[{"role": "user", "content": f"Security review:\n```python\n{state['code']}\n```"}]
    )
    return {"security_findings": resp.content[0].text}

# ─── NODE 4: Style Checker (Haiku — fast) ────────
def check_style(state: ReviewState) -> dict:
    resp = claude.messages.create(
        model="claude-3-5-haiku-20241022", max_tokens=512,
        system="""Check Python style against PEP 8 and Google style guide.
Focus on: naming conventions, docstrings, type hints, complexity,
magic numbers, comment quality, function length.""",
        messages=[{"role": "user", "content": state["code"]}]
    )
    return {"style_findings": resp.content[0].text}

# ─── NODE 5: Synthesizer — structured final report ─
def synthesize_report(state: ReviewState) -> dict:
    resp = claude.messages.create(
        model="claude-sonnet-4-5", max_tokens=2048,
        tools=[{
            "name": "code_review_report",
            "description": "Generate structured code review report",
            "input_schema": {
                "type": "object",
                "properties": {
                    "overall_score": {"type": "integer", "minimum": 0, "maximum": 100},
                    "critical_issues": {"type": "array", "items": {"type": "string"}},
                    "security_risks": {"type": "array", "items": {"type": "string"}},
                    "recommendations": {"type": "array", "items": {"type": "string"}},
                    "approve": {"type": "boolean"}
                },
                "required": ["overall_score", "critical_issues", "approve"]
            }
        }],
        tool_choice={"type": "tool", "name": "code_review_report"},
        messages=[{"role": "user", "content": f"""Synthesize these reviews into a final report:

BUG FINDINGS:
{state['bug_findings']}

SECURITY FINDINGS:
{state['security_findings']}

STYLE FINDINGS:
{state['style_findings']}"""}]
    )
    return {"final_report": resp.content[0].input}

# ─── BUILD THE GRAPH ─────────────────────────────
builder = StateGraph(ReviewState)
for name, fn in [("retrieve", retrieve_practices), ("bugs", detect_bugs),
                   ("security", analyze_security), ("style", check_style),
                   ("synthesize", synthesize_report)]:
    builder.add_node(name, fn)

builder.add_edge(START, "retrieve")
builder.add_edge("retrieve", "bugs")
# Security and style run in PARALLEL after bugs
builder.add_edge("retrieve", "security")
builder.add_edge("retrieve", "style")
builder.add_edge(["bugs", "security", "style"], "synthesize")
builder.add_edge("synthesize", END)

with SqliteSaver.from_conn_string("./reviews.db") as mem:
    graph = builder.compile(checkpointer=mem)

# Usage
result = graph.invoke(
    {"code": open("my_module.py").read(), "language": "python"},
    {"configurable": {"thread_id": "review-1"}}
)
report = result["final_report"]
print(f"Score: {report['overall_score']}/100 | Approve: {report['approve']}")
print(f"Critical issues: {len(report['critical_issues'])}")
💡

Model Selection in the Capstone

Notice the deliberate model choices: Bug Detector and Style Checker use claude-3-5-haiku-20241022 — fast and cheap, perfect for tasks with clear, constrained outputs. Security Analyzer uses claude-sonnet-4-5 — its superior reasoning catches subtle vulnerabilities that Haiku might miss. The Synthesizer also uses Sonnet to produce a well-structured, nuanced final report. This three-tier cost optimization can reduce your API costs by 60-70%.

Evaluation Suite for the Capstone

Your agent isn't done until it's evaluated. Here's a comprehensive test suite that validates the bug detector, security analyzer, and final report quality.

test_code_review_agent.py
import pytest
from code_review_agent import graph

# Golden dataset: code snippets with known issues
KNOWN_BUG = """
def divide(a, b):
    return a / b  # Bug: no ZeroDivisionError handling
"""

KNOWN_SQLI = """
def get_user(user_id):
    query = f"SELECT * FROM users WHERE id = {user_id}"  # SQLi
    return db.execute(query)
"""

CLEAN_CODE = """
def calculate_average(numbers: list[float]) -> float:
    '''Calculate arithmetic mean with input validation.'''
    if not numbers:
        raise ValueError("Cannot average empty list")
    return sum(numbers) / len(numbers)
"""

class TestBugDetector:
    def test_catches_division_by_zero(self):
        result = graph.invoke({"code": KNOWN_BUG, "language": "python"})
        assert "ZeroDivisionError" in result["bug_findings"] or \
               "zero" in result["bug_findings"].lower()

    def test_clean_code_no_critical_bugs(self):
        result = graph.invoke({"code": CLEAN_CODE, "language": "python"})
        report = result["final_report"]
        assert len(report.get("critical_issues", [])) == 0
        assert report["overall_score"] >= 70

class TestSecurityAnalyzer:
    def test_catches_sql_injection(self):
        result = graph.invoke({"code": KNOWN_SQLI, "language": "python"})
        findings = result["security_findings"].lower()
        assert "sql" in findings or "injection" in findings

    def test_sqli_not_approved(self):
        result = graph.invoke({"code": KNOWN_SQLI, "language": "python"})
        assert result["final_report"]["approve"] == False

class TestReportStructure:
    def test_report_has_required_fields(self):
        result = graph.invoke({"code": CLEAN_CODE, "language": "python"})
        report = result["final_report"]
        for field in ["overall_score", "critical_issues", "approve"]:
            assert field in report
        assert 0 <= report["overall_score"] <= 100

# Run: pytest test_code_review_agent.py -v --tb=short
