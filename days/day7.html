Define State (TypedDict)
▼
Every field in the state TypedDict has a clear owner and purpose. Each agent writes only its own field — no contention, no race conditions.
Python
from typing import TypedDict, Annotated, Optional
from langgraph.graph.message import add_messages

class ReviewState(TypedDict):
    pr_url: str           # Input: GitHub PR URL
    diff_content: str    # Fetched by: orchestrator

    # Each agent writes ONLY its own field
    security_findings: list   # Written by: security_agent
    perf_findings: list       # Written by: performance_agent
    style_findings: list      # Written by: style_agent

2
Build Tools (Static Analysis)
▼
Tools run before the LLM agents to prime them with fast static analysis results. The LLM agents then focus on deeper reasoning, not pattern matching.
Python
import re

def check_vulnerabilities(code: str) -> list[dict]:
    """Fast regex-based vuln scan to prime the LLM agent."""
    patterns = [
        (re.compile(r'execute\s*\(\s*f["\']'), "CRITICAL", "SQL injection"),
        (re.compile(r'password\s*=\s*["\'][^"\']{6,}', re.I), "CRITICAL", "Hardcoded secret"),
        (re.compile(r'\beval\s*\('), "HIGH", "eval() usage"),
        (re.compile(r'hashlib\.md5\s*\('), "HIGH", "Weak hash (MD5)"),
    ]
    findings = []
    for i, line in enumerate(code.split("\n"), 1):
        for pat, sev, title in patterns:
            if pat.search(line):
                findings.append({"line": i, "severity": sev, "title": title})
    return findings

4
Wire with LangGraph (Parallel Nodes)
▼
Multiple edges from one source = LangGraph runs them in parallel. The critic node has 3 incoming edges — it waits for ALL three agents to complete before running (fan-in).
Python
from langgraph.graph import StateGraph, START, END

graph = StateGraph(ReviewState)
graph.add_node("orchestrator", orchestrator_node)
graph.add_node("security", security_agent)
graph.add_node("performance", performance_agent)
graph.add_node("style", style_agent)
graph.add_node("critic", critic_agent)
graph.add_node("formatter", formatter_agent)

graph.add_edge(START, "orchestrator")

# Fan-out: orchestrator -> 3 agents simultaneously
graph.add_edge("orchestrator", "security")
graph.add_edge("orchestrator", "performance")
graph.add_edge("orchestrator", "style")

# Fan-in: critic waits for all 3 to finish
graph.add_edge("security", "critic")
graph.add_edge("performance", "critic")
graph.add_edge("style", "critic")
graph.add_edge("critic", "formatter")
graph.add_edge("formatter", END)
5
Add FastAPI Endpoint
▼
Wrap the pipeline in a FastAPI service. Include a sync endpoint for testing and a streaming endpoint for production user-facing calls.
Python
from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI(title="Code Review Agent")

class ReviewRequest(BaseModel):
    pr_url: str

# Blocking endpoint (for testing)
@app.post("/review/sync")
async def review_sync(request: ReviewRequest) -> dict:
    state = await asyncio.wait_for(
        run_review_pipeline(request.pr_url),
        timeout=60.0,
    )
    return state.get("final_report", {})
6
Add Streaming (Server-Sent Events)
▼
SSE streaming lets the client see progress in real-time. Users see "Fetching PR..." and "Analysis running..." rather than staring at a spinner for 30 seconds.
Python
from fastapi.responses import StreamingResponse

@app.post("/review")
async def review_stream(request: ReviewRequest) -> StreamingResponse:
    async def event_stream():
        # Emit start event immediately
        yield f'data: {{"type": "start"}}\n\n'

        try:
            yield f'data: {{"type": "progress", "msg": "Running analysis..."}}\n\n'
            state = await run_review_pipeline(request.pr_url)
            yield f'data: {json.dumps({"type": "complete", "report": state["final_report"]})}\n\n'

        except Exception as e:
            yield f'data: {{"type": "error", "message": "{e}"}}\n\n'

    return StreamingResponse(event_stream(), media_type="text/event-stream")
